# 1. Ковровая дорожка на арене
import math
arena_radius = 6.5
carpet_a = float(input("Введите длину ковровой дорожки (в метрах): "))
carpet_b = float(input("Введите ширину ковровой дорожки (в метрах): "))
carpet_diagonal = math.sqrt(carpet_a**2 + carpet_b**2)
if carpet_diagonal <= arena_radius * 2:
    print("да")
else:
    print("нет")

# 9. Относительное расположение окружностей
import math
x1 = float(input("Введите x-координату центра первой окружности: "))
y1 = float(input("Введите y-координату центра первой окружности: "))
r1 = float(input("Введите радиус первой окружности: "))
x2 = float(input("Введите x-координату центра второй окружности: "))
y2 = float(input("Введите y-координату центра второй окружности: "))
r2 = float(input("Введите радиус второй окружности: "))
distance = math.sqrt((x1 - x2)**2 + (y1 - y2)**2)
if distance > r1 + r2:
    print("Окружности лежат одна вне другой, не касаясь")
elif distance == r1 + r2:
    print("Окружности имеют внешнее касание")
elif distance < r1 + r2 and distance > abs(r1 - r2):
    print("Окружности пересекаются")
elif distance == abs(r1 - r2):
    print("Окружности имеют внутреннее касание")
else:
    print("Одна окружность лежит внутри другой, не касаясь")

# 10. Относительное расположение прямоугольников
x1 = float(input("Введите x-координату левой верхней вершины первого прямоугольника: "))
y1 = float(input("Введите y-координату левой верхней вершины первого прямоугольника: "))
x2 = float(input("Введите x-координату правой нижней вершины первого прямоугольника: "))
y2 = float(input("Введите y-координату правой нижней вершины первого прямоугольника: "))

x3 = float(input("Введите x-координату левой верхней вершины второго прямоугольника: "))
y3 = float(input("Введите y-координату левой верхней вершины второго прямоугольника: "))
x4 = float(input("Введите x-координату правой нижней вершины второго прямоугольника: "))
y4 = float(input("Введите y-координату правой нижней вершины второго прямоугольника: "))


if (x2 < x3 or x4 < x1 or y2 > y3 or y4 > y1):
    print("Прямоугольники лежат вне друг друга, не касаясь")
elif (x2 == x3 or x4 == x1 or y2 == y3 or y4 == y1):
    print("Прямоугольники имеют касание")
elif (x1 < x4 and x3 < x2 and y1 > y4 and y3 > y2):
    print("Прямоугольники имеют пересечение")
else:
    print("Один прямоугольник лежит внутри другого")
